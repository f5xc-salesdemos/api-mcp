  -H "Authorization: APIToken ${F5XC_API_TOKEN}" \

  -H "Authorization: APIToken ${F5XC_API_TOKEN}" \
  -H "Content-Type: application/json" \

# Delete resource

  -H "Authorization: APIToken ${F5XC_API_TOKEN}"
