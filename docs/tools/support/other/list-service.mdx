| Tool | Description |

# List resources

  -H "Content-Type: application/json"

  -H "Authorization: APIToken ${F5XC_API_TOKEN}" \
  -H "Content-Type: application/json" \
  -d @list_service.json

# Delete resource

  -H "Authorization: APIToken ${F5XC_API_TOKEN}"
