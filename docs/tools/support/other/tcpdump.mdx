```bash
# List resources

  -H "Authorization: APIToken ${F5XC_API_TOKEN}" \
  -H "Content-Type: application/json"

# Get specific resource

  -H "Authorization: APIToken ${F5XC_API_TOKEN}" \

  -H "Authorization: APIToken ${F5XC_API_TOKEN}" \
  -H "Content-Type: application/json" \
  -d @tcpdump.json

# Delete resource

  -H "Authorization: APIToken ${F5XC_API_TOKEN}"
```
