```bash
# Create resource
curl -X POST "https://${TENANT}.console.ves.volterra.io/api/data/namespaces/$${PARAM}/global_log_receiver/status" \
  -H "Authorization: APIToken ${F5XC_API_TOKEN}" \
  -H "Content-Type: application/json" \
  -d @statu.json

```
